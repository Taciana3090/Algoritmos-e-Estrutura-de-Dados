Q4) Functions in C
//#include <stdio.h>
/*
Add `int max_of_four(int a, int b, int c, int d)` here.
*/


int max_of_four(int a, int b, int c, int d) {
    int ans;
    a = a > b ? a : b;
    c = c > d ? c : d;
    ans = a > c ? a : c;
    return ans;
}


int main() {
    int a, b, c, d;
    scanf("%d %d %d %d", &a, &b, &c, &d);
    int ans = max_of_four(a, b, c, d);
    printf("%d", ans);
    
    return 0;
}

Q5) Pointers in C
//#include <stdio.h>


void update(int *a,int *b) {
    // Complete this function    
    int t = *a;
    *a = *a + *b;
    *b = t > *b ? t - *b : *b - t;
}


int main() {
    int a, b;
    int *pa = &a, *pb = &b;
    
    scanf("%d %d", &a, &b);
    update(pa, pb);
    printf("%d\n%d", a, b);


    return 0;
}
